2. ЗАВДАННЯ №1

Створити JavaScript-код, який:

  • Зберігає введене значення (використовуючи метод prompt()) у змінну value
  • Виводить це значення у Console, використовуючи шаблонний рядок
  • Перевіряє: введене значення від'ємне? Додатнє? Дорівнює нулю?
  • Відповідь виводить за допомогою alert()

2.1. Програмний код

// Збережіть введене значення (використати метод prompt()) у змінну value
let value = prompt("Введіть число:");

// Вивести це значення у Console, використовуючи шаблонний рядок
console.log(`Введене значення: ${value}`);

// Перевірити: введене значення від'ємне? Додатнє? Дорівнює нулю?
// Перетворюємо рядок на число для перевірки
let numberValue = Number(value);

// Перевірка на валідність числа
if (isNaN(numberValue)) {
    console.log("Помилка: введено не число!");
    alert("Помилка: введено не число!");
} else {
    // Відповідь вивести за допомогою alert()
    if (numberValue > 0) {
        console.log("Число додатнє");
        alert("Число додатнє");
    } else if (numberValue < 0) {
        console.log("Число від'ємне");
        alert("Число від'ємне");
    } else {
        console.log("Число дорівнює нулю");
        alert("Число дорівнює нулю");
    }
}

2.2. Пояснення коду

Основні елементи коду:

  • prompt("Введіть число:") - метод для отримання введеного користувачем значення. Повертає рядок або null, якщо користувач натиснув "Скасувати"
  • console.log(`Введене значення: ${value}`) - виведення значення в консоль браузера з використанням шаблонного рядка (template literal). Шаблонні рядки дозволяють вставляти змінні через ${}
  • Number(value) - перетворення рядка на число. Якщо перетворення неможливе, повертає NaN
  • isNaN(numberValue) - перевірка, чи є значення NaN (Not a Number). Використовується для валідації введених даних
  • if-else if-else - умовні оператори для перевірки знака числа (додатнє, від'ємне, нуль)
  • alert() - метод для виведення повідомлення в діалоговому вікні браузера

3. ЗАВДАННЯ №2

Створити JavaScript-код, який:

  • Запитує значення у користувача через prompt() (число від 1 до 4)
  • Використовує switch-case для визначення пори року
  • Має дефолтне значення для некоректних введень
  • Виводить результат в консоль через console.log()

3.1. Програмний код

// Запитуємо значення у користувача через prompt
let number = prompt("Введіть число від 1 до 4:");

// Змінна для збереження результату
let result;

// Використовуємо switch-case для визначення пори року
switch (number) {
    case '1':
        result = 'зима';
        break;
    case '2':
        result = 'весна';
        break;
    case '3':
        result = 'літо';
        break;
    case '4':
        result = 'осінь';
        break;
    default:
        result = 'невідома пора року';
        break;
}

// Виводимо значення змінної result в консоль
console.log(`Результат: ${result}`);

3.2. Пояснення коду

Основні елементи коду:

  • prompt("Введіть число від 1 до 4:") - метод для отримання введеного користувачем значення. Повертає рядок
  • switch (number) - оператор switch використовується для вибору одного з багатьох блоків коду для виконання. Більш зручний для множинних перевірок, ніж if-else
  • case '1': - перевіряє, чи значення number дорівнює '1'. Якщо так, виконується код після двокрапки
  • break; - виходить з блоку switch, запобігаючи виконанню наступних case. Без break виконаються всі наступні case (fall-through)
  • default: - виконується, якщо жоден case не відповідає значенню. Використовується для обробки некоректних або неочікуваних значень
  • console.log(`Результат: ${result}`) - виведення результату в консоль браузера з використанням шаблонного рядка

Створити JavaScript-код для системи авторизації, який:

  • Об'являє та ініціалізує логіни та паролі для Admin та User
  • Запитує логін у користувача через prompt()
  • Виводить введений логін в консоль браузера, використовуючи шаблонний рядок
  • Перевіряє введене значення:
    • Якщо "Admin" - запитує пароль, при вдалій ідентифікації виводить "Hello, Admin"
    • Якщо нічого не ввели або натиснули Cancel - виводить "Cancelled"
    • В іншому випадку - виводить "I don't know you"

4.1. Програмний код

// Об'являємо та ініціалізуємо логіни та паролі
const Admin = 'Admin';
const adminPassword = 'admin123';
const User = 'User';
const userPassword = 'user123';

// Запитуємо логін у користувача через prompt
let login = prompt("Введіть логін:");

// Виводимо результат в консоль браузера, використовуючи шаблонний рядок
console.log(`Введений логін: ${login}`);

// Перевірка введеного користувачем значення
if (login === null || login.trim() === '') {
    // Якщо нічого не ввели або натиснули Cancel
    console.log("Cancelled");
    alert("Cancelled");
} else if (login.trim() === Admin) {
    // Якщо відвідувач вводить "Admin", то prompt запитує пароль
    let password = prompt("Введіть пароль:");
    
    // У випадку вдалої ідентифікації в alert вивести «Hello, Admin»
    if (password === adminPassword) {
        console.log("Hello, Admin");
        alert("Hello, Admin");
    } else {
        console.log("I don't know you");
        alert("I don't know you");
    }
} else {
    // В іншому випадку вивести в alert рядок "I don't know you"
    console.log("I don't know you");
    alert("I don't know you");
}

4.2. Пояснення коду

Основні елементи коду:

  • const Admin = 'Admin' - об'явлення та ініціалізація константи з логіном Admin
  • const adminPassword = 'admin123' - об'явлення та ініціалізація константи з паролем для Admin
  • prompt("Введіть логін:") - метод для отримання логіну від користувача. Повертає рядок або null, якщо натиснуто Cancel
  • console.log(`Введений логін: ${login}`) - виведення логіну в консоль з використанням шаблонного рядка
  • login === null - перевірка, чи користувач натиснув Cancel (prompt повертає null)
  • login.trim() === '' - перевірка, чи введено порожній рядок (trim() видаляє пробіли на початку та в кінці)
  • login.trim() === Admin - перевірка, чи введений логін відповідає Admin (з видаленням пробілів)
  • password === adminPassword - перевірка правильності введеного пароля
  • alert() - метод для виведення повідомлень користувачу в діалоговому вікні

6. ЗАВДАННЯ №5

Створити функцію для перевірки повідомлень на заборонені слова, яка:

  • Оголошує функцію checkForSpam з параметром message (рядок)
  • Перевіряє рядок на вміст заборонених слів "spam" та "sale"
  • Працює незалежно від регістру букв (SPAM, sAlE, SpAm тощо)
  • Повертає true, якщо знайдено заборонене слово
  • Повертає false, якщо заборонених слів немає

6.1. Програмний код

// Оголошуємо функцію checkForSpam
function checkForSpam(message) {
    // Перетворюємо рядок на нижній регістр для порівняння незалежно від регістру
    let lowerMessage = message.toLowerCase();
    
    // Перевіряємо наявність заборонених слів
    if (lowerMessage.includes('spam') || lowerMessage.includes('sale')) {
        return true;
    } else {
        return false;
    }
}

// Приклади використання функції
console.log(checkForSpam("Latest technology news")); // false
console.log(checkForSpam("JavaScript weekly newsletter")); // false
console.log(checkForSpam("Get best sale offers now!")); // true
console.log(checkForSpam("Amazing SalE, only today!")); // true
console.log(checkForSpam("Trust me, this is not a spam message")); // true
console.log(checkForSpam("Get rid of sPaM emails. Our book in on sale!")); // true
console.log(checkForSpam("[SPAM] How to earn fast money?")); // true

6.2. Пояснення коду

Основні елементи коду:

  • function checkForSpam(message) - оголошення функції з одним параметром message (рядок)
  • message.toLowerCase() - метод для перетворення рядка на нижній регістр. Дозволяє перевіряти слова незалежно від того, як вони написані (SPAM, SpAm, spam - все стане "spam")
  • lowerMessage.includes('spam') - метод includes() перевіряє, чи містить рядок підрядок 'spam'. Повертає true, якщо знайдено, false - якщо ні
  • || - логічний оператор "АБО" (OR). Вираз повертає true, якщо хоча б одна з умов істинна
  • return true - повертає бульове значення true, якщо знайдено заборонене слово
  • return false - повертає бульове значення false, якщо заборонених слів не знайдено
  • Бульові значення (boolean) - тип даних, який може мати тільки два значення: true (істина) або false (хибність)

7. ЗАВДАННЯ №6

Створити функцію для фільтрації масиву чисел, яка:

  • Оголошує функцію filterArray з двома параметрами:
    • numbers - масив чисел
    • value - значення для порівняння
  • Створює порожній масив для збереження підходящих чисел
  • Використовує цикл для ітерації кожного елемента масиву
  • Використовує умовний оператор if для перевірки та додавання елементів
  • Повертає новий масив лише з тих чисел, які більші за value

7.1. Програмний код

// Оголошуємо функцію filterArray
function filterArray(numbers, value) {
    // Створюємо порожній масив, у який будемо додавати підходящі числа
    let filteredArray = [];
    
    // Використовуємо цикл для ітерації кожного елемента масиву numbers
    for (let i = 0; i < numbers.length; i++) {
        // Використовуємо умовний оператор if для перевірки кожного елемента
        if (numbers[i] > value) {
            // Додаємо елемент до нового масиву, якщо він більший за value
            filteredArray.push(numbers[i]);
        }
    }
    
    // Повертаємо новий масив з підходящими числами
    return filteredArray;
}

7.2. Пояснення коду

Основні елементи коду:

  • function filterArray(numbers, value) - оголошення функції з двома параметрами: масивом чисел та значенням для порівняння
  • let filteredArray = [] - створення порожнього масиву для збереження відфільтрованих чисел
  • for (let i = 0; i < numbers.length; i++) - цикл for для ітерації по всіх елементах масиву. i - індекс поточного елемента, починається з 0 і збільшується до numbers.length - 1
  • numbers[i] - доступ до елемента масиву за індексом i
  • numbers.length - властивість масиву, що повертає кількість елементів у масиві
  • if (numbers[i] > value) - умовний оператор для перевірки, чи поточний елемент більший за value
  • filteredArray.push(numbers[i]) - метод push() додає елемент до кінця масиву
  • return filteredArray - повертає новий масив з відфільтрованими числами
  • Масиви (arrays) - структура даних для зберігання колекції елементів. Елементи мають індекси, починаючи з 0

Створити програму для роботи з масивами та сортуванням, яка:

8.1. Програмний код

// Функція для знаходження максимального елемента масиву
function findMax(array) {
    let max = array[0];
    for (let i = 1; i < array.length; i++) {
        if (array[i] > max) {
            max = array[i];
        }
    }
    return max;
}

// Функція для побудови масиву B: bі = max * ai
function buildArrayB(arrayA) {
    let max = findMax(arrayA);
    let arrayB = [];
    
    for (let i = 0; i < arrayA.length; i++) {
        arrayB.push(max * arrayA[i]);
    }
    
    return arrayB;
}

// Функція для сортування масиву за зменшенням методом вставки
function insertionSortDescending(array) {
    let sortedArray = [...array]; // Створюємо копію масиву
    
    for (let i = 1; i < sortedArray.length; i++) {
        let current = sortedArray[i];
        let j = i - 1;
        
        // Переміщуємо елементи, які менші за current, на одну позицію вправо
        while (j >= 0 && sortedArray[j] < current) {
            sortedArray[j + 1] = sortedArray[j];
            j--;
        }
        
        sortedArray[j + 1] = current;
    }
    
    return sortedArray;
}

// Функція для виведення масиву
function printArray(array, name) {
    console.log(`${name}: [${array.join(', ')}]`);
}

8.2. Пояснення коду

Основні елементи коду:

  • function findMax(array) - функція для знаходження максимального елемента масиву. Проходить по всіх елементах та знаходить найбільший
  • function buildArrayB(arrayA) - функція для побудови масиву B. Кожен елемент bі = max * ai, де max - максимальний елемент масиву A
  • function insertionSortDescending(array) - функція для сортування масиву за зменшенням методом вставки. Алгоритм працює так: береться елемент та вставляється в правильну позицію серед вже відсортованих елементів
  • [...array] - оператор spread для створення копії масиву, щоб не змінювати оригінальний
  • while (j >= 0 && sortedArray[j] < current) - цикл while для знаходження правильної позиції для вставки елемента. Умова sortedArray[j] < current означає сортування за зменшенням
  • function printArray(array, name) - функція для виведення масиву в консоль у зручному форматі
  • array.join(', ') - метод join() об'єднує елементи масиву в рядок з роздільником ", "
  • parseInt() - функція для перетворення рядка на ціле число
  • parseFloat() - функція для перетворення рядка на число з плаваючою комою
  • Метод вставки (Insertion Sort) - алгоритм сортування, який працює шляхом побудови відсортованого масиву по одному елементу за раз

// Функція для генерації випадкового числа в діапазоні від min до max
function getRandomNumber(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

// Функція для створення двовимірного масиву з випадковими числами
function create2DArray(rows, cols, minValue, maxValue) {
    let array = [];
    for (let i = 0; i < rows; i++) {
        let row = [];
        for (let j = 0; j < cols; j++) {
            row.push(getRandomNumber(minValue, maxValue));
        }
        array.push(row);
    }
    return array;
}

// Функція для виведення двовимірного масиву в консоль
function print2DArray(array, name) {
    console.log(`${name}:`);
    for (let i = 0; i < array.length; i++) {
        console.log(` [${array[i].join(', ')}]`);
    }
}

// Функція для розділення чисел на додатні та від'ємні
function separateNumbers(array2D) {
    let positiveArray = [];
    let negativeArray = [];
    
    for (let i = 0; i < array2D.length; i++) {
        for (let j = 0; j < array2D[i].length; j++) {
            if (array2D[i][j] > 0) {
                positiveArray.push(array2D[i][j]);
            } else if (array2D[i][j] < 0) {
                negativeArray.push(array2D[i][j]);
            }
        }
    }
    
    return { positive: positiveArray, negative: negativeArray };
}

9.2. Пояснення коду

Основні елементи коду:

  • function getRandomNumber(min, max) - функція для генерації випадкового числа в заданому діапазоні. Використовує Math.random() та Math.floor()
  • Math.random() - метод, що повертає випадкове число від 0 до 1 (не включно)
  • Math.floor() - метод для округлення числа вниз до найближчого цілого
  • function create2DArray(rows, cols, minValue, maxValue) - функція для створення двовимірного масиву (масив масивів) з випадковими числами
  • Двовимірний масив - структура даних, що представляє собою масив масивів. Кожен елемент є масивом
  • Вкладені цикли for - використання циклу всередині циклу для обробки двовимірних структур даних
  • function print2DArray(array, name) - функція для виведення двовимірного масиву в консоль у зручному форматі
  • function separateNumbers(array2D) - функція для розділення чисел з двовимірного масиву на додатні та від'ємні
  • Деструктуризація об'єкта - використання синтаксису { positive: positiveArray, negative: negativeArray } для отримання значень з об'єкта
  • positiveArray[2] = negativeValue - заміна третього елемента (індекс 2) у масиві на нове значення

// Функція для додавання цифри
function appendNumber(number) {
    if (shouldResetDisplay) {
        currentInput = '0';
        shouldResetDisplay = false;
    }
    
    if (currentInput === '0') {
        currentInput = number.toString();
    } else {
        currentInput += number.toString();
    }
    updateDisplay();
}

// Функція для обчислення результату
function calculate() {
    let result;
    const prev = parseFloat(previousInput);
    const current = parseFloat(currentInput);
    
    if (isNaN(prev) || isNaN(current)) return;
    
    switch (operation) {
        case '+':
            result = prev + current;
            break;
        case '-':
            result = prev - current;
            break;
        case '*':
            result = prev * current;
            break;
        case '/':
            if (current === 0) {
                alert('Помилка: ділення на нуль!');
                clear();
                return;
            }
            result = prev / current;
            break;
        case '%':
            result = prev % current;
            break;
        case '^':
            result = Math.pow(prev, current);
            break;
        default:
            return;
    }
    
    currentInput = result.toString();
    previousInput = '';
    operation = null;
    shouldResetDisplay = true;
    updateDisplay();
}

// Функція для обчислення квадратного кореня
function calculateSquareRoot() {
    const current = parseFloat(currentInput);
    
    if (isNaN(current)) return;
    
    if (current < 0) {
        alert('Помилка: неможливо обчислити корінь з від\'ємного числа!');
        return;
    }
    
    currentInput = Math.sqrt(current).toString();
    shouldResetDisplay = true;
    updateDisplay();
}
